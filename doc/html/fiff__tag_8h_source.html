693 inline FiffChInfo FiffTag::toChInfo() const
694 {
695  FiffChInfo p_FiffChInfo;
696 
697  if(this->isMatrix() || this->getType() != FIFFT_CH_INFO_STRUCT || this->data == NULL)
698  return p_FiffChInfo;
699  else
700  {
701  qint32* t_pInt32 = static_cast< qint32* >(this->data);
702  p_FiffChInfo.scanno = t_pInt32[0];
703  p_FiffChInfo.logno = t_pInt32[1];
704  p_FiffChInfo.kind = t_pInt32[2];
705  float* t_pFloat = static_cast< float* >(this->data);
706  p_FiffChInfo.range = t_pFloat[3];
707  p_FiffChInfo.cal = t_pFloat[4];
708  p_FiffChInfo.coil_type = t_pInt32[5];
709 
710  //
711  // Read the coil coordinate system definition
712  //
713  qint32 count = 0;
714  qint32 r, c;
715  for (r = 0; r < 12; ++r) {
716  p_FiffChInfo.loc(r,0) = t_pFloat[6+r];
717  }
718 
719  p_FiffChInfo.coord_frame = FIFFV_COORD_UNKNOWN;
720 
721  //
722  // Convert loc into a more useful format
723  //
724  if (p_FiffChInfo.kind == FIFFV_MEG_CH || p_FiffChInfo.kind == FIFFV_REF_MEG_CH)
725  {
726  p_FiffChInfo.coil_trans.setIdentity(4,4);
727  for (r = 0; r < 3; ++r) {
728  p_FiffChInfo.coil_trans(r,3) = p_FiffChInfo.loc(r,0);
729  for (c = 0; c < 3; ++c) {
730  p_FiffChInfo.coil_trans(c,r) = p_FiffChInfo.loc(3+count,0);//its transposed stored (r and c are exchanged)
731  ++count;
732  }
733  }
734  p_FiffChInfo.coord_frame = FIFFV_COORD_DEVICE;
735  }
736  else if (p_FiffChInfo.kind == FIFFV_EEG_CH)
737  {
738  if (p_FiffChInfo.loc.block(3,0,3,1).norm() > 0)
739  {
740  p_FiffChInfo.eeg_loc.block(0,0,3,1) = p_FiffChInfo.loc.block(0,0,3,1);
741  p_FiffChInfo.eeg_loc.block(0,1,3,1) = p_FiffChInfo.loc.block(3,0,3,1);
742  }
743  else
744  {
745  p_FiffChInfo.eeg_loc.block(0,0,3,1) = p_FiffChInfo.loc.block(0,0,3,1);
746  }
747  p_FiffChInfo.coord_frame = FIFFV_COORD_HEAD;
748  }
749  //
750  // Unit and exponent
751  //
752  p_FiffChInfo.unit = t_pInt32[18];
753  p_FiffChInfo.unit_mul = t_pInt32[19];
754 
755  //
756  // Handle the channel name
757  //
758  char* orig = static_cast< char* >(this->data);
759  p_FiffChInfo.ch_name = QString::fromAscii(orig + 80);
760 
761  return p_FiffChInfo;
