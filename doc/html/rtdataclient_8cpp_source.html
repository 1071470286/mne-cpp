104 FiffInfo RtDataClient::readInfo()
105 {
106  FiffInfo p_FiffInfo;
107  bool t_bReadMeasBlockStart = false;
108  bool t_bReadMeasBlockEnd = false;
109 
110  FiffStream t_fiffStream(this);
111  //
112  // Find the start
113  //
114  FiffTag* t_pTag = NULL;
115  while(!t_bReadMeasBlockStart)
116  {
117  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
118  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_MEAS_INFO)
119  {
120  printf("FIFF_BLOCK_START FIFFB_MEAS_INFO\n");
121  t_bReadMeasBlockStart = true;
122  }
123  }
124 
125  //
126  // Parse until the endblock
127  //
128 
129  bool dev_head_t_read = false;
130  bool ctf_head_t_read = false;
131 
132  while(!t_bReadMeasBlockEnd)
133  {
134  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
135  //
136  // megacq parameters
137  //
138  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_DACQ_PARS)
139  {
140  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_DACQ_PARS)
141  {
142  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
143  if(t_pTag->kind == FIFF_DACQ_PARS)
144  p_FiffInfo.acq_pars = t_pTag->toString();
145  else if(t_pTag->kind == FIFF_DACQ_STIM)
146  p_FiffInfo.acq_stim = t_pTag->toString();
147  }
148  }
149  //
150  // Coordinate transformations if the HPI result block was not there
151  //
152  if (t_pTag->kind == FIFF_COORD_TRANS)
153  {
154  if (!dev_head_t_read)
155  {
156  p_FiffInfo.dev_head_t = t_pTag->toCoordTrans();
157  dev_head_t_read = true;
158  }
159  else if (!ctf_head_t_read)
160  {
161  p_FiffInfo.ctf_head_t = t_pTag->toCoordTrans();
162  ctf_head_t_read = true;
163  }
164  }
165  //
166  // Polhemus data
167  //
168  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_ISOTRAK)
169  {
170  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_ISOTRAK)
171  {
172  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
173 
174  if(t_pTag->kind == FIFF_DIG_POINT)
175  p_FiffInfo.dig.append(t_pTag->toDigPoint());
176  }
177  }
178  //
179  // Projectors
180  //
181  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_PROJ)
182  {
183  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_PROJ)
184  {
185  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
186  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_PROJ_ITEM)
187  {
188  FiffProj proj;
189  qint32 countProj = p_FiffInfo.projs.size();
190  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_PROJ_ITEM)
191  {
192  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
193  switch (t_pTag->kind)
194  {
195  case FIFF_NAME: // First proj -> Proj is created
196  proj = FiffProj();
197  p_FiffInfo.projs.append(proj);
198  p_FiffInfo.projs[countProj].desc = t_pTag->toString();
199  break;
200  case FIFF_PROJ_ITEM_KIND:
201  p_FiffInfo.projs[countProj].kind = *(t_pTag->toInt());
202  break;
203  case FIFF_NCHAN: // First data -> FiffNamedMatrix is created
204 // p_FiffInfo.projs[countProj].data = FiffNamedMatrix();//obsolete
205  p_FiffInfo.projs[countProj].data->ncol = *(t_pTag->toInt());
206  break;
207  case FIFF_PROJ_ITEM_NVEC:
208  p_FiffInfo.projs[countProj].data->nrow = *(t_pTag->toInt());
209  break;
210  case FIFF_MNE_PROJ_ITEM_ACTIVE:
211  p_FiffInfo.projs[countProj].active = *(t_pTag->toInt());
212  break;
213  case FIFF_PROJ_ITEM_CH_NAME_LIST:
214  p_FiffInfo.projs[countProj].data->col_names = FiffStream::split_name_list(t_pTag->toString());
215  break;
216  case FIFF_PROJ_ITEM_VECTORS:
217  p_FiffInfo.projs[countProj].data->data = t_pTag->toFloatMatrix();
218  break;
219  }
220  }
221  }
222  }
223  }
224  //
225  // CTF compensation info
226  //
227  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_MNE_CTF_COMP)
228  {
229  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_MNE_CTF_COMP)
230  {
231  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
232  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_MNE_CTF_COMP_DATA)
233  {
234  FiffCtfComp comp;
235  qint32 countComp = p_FiffInfo.comps.size();
236  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_MNE_CTF_COMP_DATA)
237  {
238  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
239  switch (t_pTag->kind)
240  {
241  case FIFF_MNE_CTF_COMP_KIND: //First comp -> create comp
242  comp = FiffCtfComp();
243  p_FiffInfo.comps.append(comp);
244  p_FiffInfo.comps[countComp].ctfkind = *(t_pTag->toInt());
245  break;
246  case FIFF_MNE_CTF_COMP_CALIBRATED:
247  p_FiffInfo.comps[countComp].save_calibrated = *(t_pTag->toInt());
248  break;
249 // case FIFF_MNE_CTF_COMP_DATA:
250 // p_FiffInfo.comps[countComp]->data = *(t_pTag->toNamedMatrix());
251 // break;
252  }
253  }
254  }
255  }
256  }
257  //
258  // Bad channels
259  //
260  if(t_pTag->kind == FIFF_BLOCK_START && *(t_pTag->toInt()) == FIFFB_MNE_BAD_CHANNELS)
261  {
262  while(t_pTag->kind != FIFF_BLOCK_END || *(t_pTag->toInt()) != FIFFB_MNE_BAD_CHANNELS)
263  {
264  FiffTag::read_rt_tag(&t_fiffStream, t_pTag);
265  if(t_pTag->kind == FIFF_MNE_CH_NAME_LIST)
266  p_FiffInfo.bads = FiffStream::split_name_list(t_pTag->data());
267  }
268  }
269  //
270  // General
271  //
272  switch(t_pTag->kind)
273  {
274  case FIFF_SFREQ:
275  p_FiffInfo.sfreq = *(t_pTag->toFloat());
276  break;
277  case FIFF_HIGHPASS:
278  p_FiffInfo.highpass = *(t_pTag->toFloat());
279  break;
280  case FIFF_LOWPASS:
281  p_FiffInfo.lowpass = *(t_pTag->toFloat());
282  break;
283  case FIFF_NCHAN:
284  p_FiffInfo.nchan = *(t_pTag->toInt());
285  break;
286  case FIFF_MEAS_DATE:
287  p_FiffInfo.meas_date[0] = t_pTag->toInt()[0];
288  p_FiffInfo.meas_date[1] = t_pTag->toInt()[1];
289  break;
290  }
291 
292  if (t_pTag->kind == FIFF_CH_INFO)
293  p_FiffInfo.chs.append(t_pTag->toChInfo());
294 
295  // END MEAS
296  if(t_pTag->kind == FIFF_BLOCK_END && *t_pTag->toInt() == FIFFB_MEAS_INFO)
297  {
298  printf("FIFF_BLOCK_END FIFFB_MEAS_INFO\n");
299  t_bReadMeasBlockEnd = true;
300  }
301  }
302 
303  //
304  // Add the channel information and make a list of channel names
305  // for convenience
306  //
307  for (qint32 c = 0; c < p_FiffInfo.nchan; ++c)
308  p_FiffInfo.ch_names << p_FiffInfo.chs[c].ch_name;
309 
310  return p_FiffInfo;
311 }
