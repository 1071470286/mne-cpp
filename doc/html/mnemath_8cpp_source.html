201 SparseMatrix<double>* MNEMath::make_block_diag(const MatrixXd* A, qint32 n)
202 {
203 
204  qint32 ma = A->rows();
205  qint32 na = A->cols();
206  float bdn = ((float)na)/n; // number of submatrices
207 
208 // std::cout << std::endl << "ma " << ma << " na " << na << " bdn " << bdn << std::endl;
209 
210  if(bdn - floor(bdn))
211  {
212  printf("Width of matrix must be even multiple of n\n");
213  return NULL;
214  }
215 
216  SparseMatrix<double>* bd = new SparseMatrix<double>((int)floor((float)ma*bdn+0.5),na);
217 
218  qint32 current_col, current_row, i, r, c;
219  for(i = 0; i < bdn; ++i)
220  {
221  current_col = i * n;
222  current_row = i * ma;
223 
224  for(r = 0; r < ma; ++r)
225  for(c = 0; c < n; ++c)
226  bd->insert(r+current_row,c+current_col) = (*A)(r, c+current_col);
227  }
228  return bd;
229 }

234 qint32 MNEMath::rank(MatrixXd& A, double tol)
235 {
236  JacobiSVD<MatrixXd> t_svdA(A);//U and V are not computed
237  VectorXd s = t_svdA.singularValues();
238  double t_dMax = s.maxCoeff();
239  t_dMax *= tol;
240  int sum = 0;
241  for(qint32 i = 0; i < s.size(); ++i)
242  sum += s[i] > t_dMax ? 1 : 0;
243  return sum;
244 }
