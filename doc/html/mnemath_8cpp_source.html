154 SparseMatrix<double>* MNEMath::make_block_diag(const MatrixXd* A, qint32 n)
155 {
156 
157  qint32 ma = A->rows();
158  qint32 na = A->cols();
159  float bdn = ((float)na)/n; // number of submatrices
160 
161 // std::cout << std::endl << "ma " << ma << " na " << na << " bdn " << bdn << std::endl;
162 
163  if(bdn - floor(bdn))
164  {
165  printf("Width of matrix must be even multiple of n\n");
166  return NULL;
167  }
168 
169  SparseMatrix<double>* bd = new SparseMatrix<double>((int)floor((float)ma*bdn+0.5),na);
170 
171  qint32 current_col, current_row, i, r, c;
172  for(i = 0; i < bdn; ++i)
173  {
174  current_col = i * n;
175  current_row = i * ma;
176 
177  for(r = 0; r < ma; ++r)
178  for(c = 0; c < n; ++c)
179  bd->insert(r+current_row,c+current_col) = (*A)(r, c+current_col);
180  }
181  return bd;
182 }
