157  dMaxValue = 1e-3f;
158  if(m_scaleMap.contains(FIFFV_EOG_CH))
159  dMaxValue = m_scaleMap[FIFFV_EOG_CH];
160  break;
161  }
162  case FIFFV_STIM_CH: {
163  dMaxValue = 5;
164  if(m_scaleMap.contains(FIFFV_STIM_CH))
165  dMaxValue = m_scaleMap[FIFFV_STIM_CH];
166  break;
167  }
168  case FIFFV_MISC_CH: {
169  dMaxValue = 1e-3f;
170  if(m_scaleMap.contains(FIFFV_MISC_CH))
171  dMaxValue = m_scaleMap[FIFFV_MISC_CH];
172  break;
173  }
174  }
175 
176  //Plot averaged data
177  QRectF boundingRect = this->boundingRect();
178  double dScaleY = (boundingRect.height()*10)/(2*dMaxValue);
179  QPointF qSamplePosition;
180 
181  //do for all currently stored evoked set data
182  for(int dataIndex = 0; dataIndex<m_lAverageData.size(); dataIndex++) {
183  //plot data from averaged data m_lAverageData with the calculated downsample factor

186

190  if(dsFactor == 0)
191  dsFactor = 1;
