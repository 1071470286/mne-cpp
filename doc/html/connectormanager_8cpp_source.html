406 QByteArray ConnectorManager::setActiveConnector(qint32 ID)
407 {
408  QByteArray p_blockClientList;

410 
411  if(ID != getActiveConnector()->getConnectorID())

415  for( ; it != s_vecConnectors.end(); ++it)

418

423

426  this->disconnectActiveConnector();
