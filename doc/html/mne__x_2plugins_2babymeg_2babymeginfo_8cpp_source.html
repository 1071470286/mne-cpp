221  qDebug()<<"[babyMEGinfo] chnNum:" << chnNum << "Data Length" <<dataLength<<"sampling rate"<<sfreq;
222  //qDebug()<<"cmdstr"<<cmdstr;
223  // Start to acquire the channel's name and channel's scale
224  MGH_LM_Get_Channel_Info(cmdstr);
225 
226  }
227  else
228  {
229  chnNum = 464;
230  dataLength = 5000;
231  sfreq = 10000;
232  }
233 
234  // Parameters
235  m_FiffInfo.file_id.version = 0; //ToDo
236 
237  m_FiffInfo.meas_date[0] = 0;
238  m_FiffInfo.meas_date[1] = 0;
239  m_FiffInfo.sfreq = sfreq;
240  m_FiffInfo.highpass = 0;

242  m_FiffInfo.acq_pars = QString("BabyMEG");
243  m_FiffInfo.acq_stim = QString("");
244  m_FiffInfo.filename = QString("");
245  m_FiffInfo.meas_id.version = 1;
246  m_FiffInfo.nchan = chnNum; //464;
247  m_FiffInfo.dev_head_t.from =4;
248  m_FiffInfo.dev_head_t.to =1;
249 
250  //set the identified matrix
251  for (int li=0;li<4;li++)
252  for(int lj=0;lj<4;lj++)
253  if (li==lj) m_FiffInfo.dev_head_t.trans(li,lj) = 1.0f;
254  else m_FiffInfo.dev_head_t.trans(li,lj) = 0.0f;
255 
256  //MEG
257  for(qint32 i = 0; i < chnNum; i++)
258  {
259  FiffChInfo t_ch;
260 
261  t_ch.ch_name = lm_ch_names.at(i); //QString("MEG%1").arg(i);

268 
269  //qDebug()<<i<<":="<<t_ch.ch_name<<","<<t_ch.range<<","<<t_ch.cal;
270 
271  t_ch.loc.setZero(12,1);
272 
273  //set loc

275  t_ch.loc(1,0) = lm_ch_pos2.at(i).toDouble();
276  t_ch.loc(2,0) = lm_ch_pos3.at(i).toDouble();
277  t_ch.loc(3,0) = lm_ch_pos4.at(i).toDouble();
278  t_ch.loc(4,0) = lm_ch_pos5.at(i).toDouble();

286 
287  //qDebug()<<t_ch.loc(0,0)<<t_ch.loc(1,0)<<t_ch.loc(2,0);
288 
289  int type = lm_ch_coiltype.at(i).toInt();
290  int ntype = 0;
291 
292  if (type == FIFFV_COIL_BABY_MAG) //inner layer MEG
293  ntype = 1;
294  else if (type == FIFFV_COIL_BABY_REF_MAG)

335  t_ch.unit_mul = FIFF_UNITM_NONE;
336  t_ch.coil_type = FIFFV_COIL_EEG;
337 
338  break;
339  default: // other unknown type sensors
340  t_ch.kind = FIFFV_MEG_CH;
341  t_ch.unit = FIFF_UNIT_T;
342  t_ch.unit_mul = FIFF_UNITM_NONE;
343  t_ch.coil_type = FIFFV_COIL_NONE;
344 
345  break;
346  }
347  m_FiffInfo.chs.append(t_ch);
348  m_FiffInfo.ch_names.append(t_ch.ch_name);

350  }
