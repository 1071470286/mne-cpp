201 void ImageSc::updateMaps()
202 {
203  if(m_pPixmapData)
204  {
205  delete m_pPixmapData;
206  m_pPixmapData = NULL;
207  }
208  if(m_pPixmapColorbar)
209  {
210  delete m_pPixmapColorbar;
211  m_pPixmapColorbar = NULL;
212  }
213 
214  if(m_matCentNormData.rows() > 0 && m_matCentNormData.cols() > 0)
215  {
216  // --Data--
217  qint32 x = m_matCentNormData.cols();
218  qint32 y = m_matCentNormData.rows();
219  qint32 i, j;
220  QImage t_qImageData(x, y, QImage::Format_RGB32);
221  for(i = 0; i < x; ++i)
222  for(j = 0; j < y; ++j)
223  t_qImageData.setPixel(i, j, pColorMapper(m_matCentNormData(j,i)));
224 
225  m_pPixmapData = new QPixmap(QPixmap::fromImage(t_qImageData));
226 
227  // --Colorbar--
228  QImage t_qImageColorbar(1, m_iColorbarGradSteps, QImage::Format_RGB32);
229 
230  double t_dQuantile = 1.0/((double)m_iColorbarGradSteps-1);
231  for(j = 0; j < m_iColorbarGradSteps; ++j)
232  {
233  QRgb t_qRgb = pColorMapper(t_dQuantile*((double)(m_iColorbarGradSteps-1-j))*1.0);
234  t_qImageColorbar.setPixel(0, j, t_qRgb);
235  }
236  m_pPixmapColorbar = new QPixmap(QPixmap::fromImage(t_qImageColorbar));
237 
238 
239  // --Scale Values--
240  m_qVecScaleValues.clear();
241 
242  double scale = pow(10, floor(log(m_dMaxValue-m_dMinValue)/log(10.0)));
243 
244  //Zero Based Scale?
245  if(m_dMaxValue > 0 && m_dMinValue < 0)
246  {
247  double quantum = floor((((m_dMaxValue-m_dMinValue)/scale)/(m_iColorbarSteps-1))*10.0)*(scale/10.0);
248  double start = 0;
249  while(m_dMinValue < (start - quantum))
250  start -= quantum;
251  //Create Steps
252  m_qVecScaleValues.push_back(start);
253  for(qint32 i = 1; i < m_iColorbarSteps-1; ++i)
254  m_qVecScaleValues.push_back(m_qVecScaleValues[i-1]+quantum);
255  }
256  else
257  {
258  double quantum = floor((((m_dMaxValue-m_dMinValue)/scale)/(m_iColorbarSteps-1))*10.0)*(scale/10.0);
259  double start = floor(((m_dMaxValue-m_dMinValue)/2.0 + m_dMinValue)/scale)*scale;
260  while(m_dMinValue < (start - quantum))
261  start -= quantum;
262  //Create Steps
263  m_qVecScaleValues.push_back(start);
264  for(qint32 i = 1; i < m_iColorbarSteps-1; ++i)
265  m_qVecScaleValues.push_back(m_qVecScaleValues[i-1]+quantum);
266  }
267  update();
268  }
269 }
