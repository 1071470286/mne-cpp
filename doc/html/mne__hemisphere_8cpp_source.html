171 bool MNEHemisphere::transform_hemisphere_to(fiff_int_t dest, FiffCoordTrans& trans)
172 {
173  if (this->coord_frame == dest)
174  {
175 // res = src;
176  return true;
177  }
178 
179  if (trans.to == this->coord_frame && trans.from == dest)
180  trans.invert_transform();
181  else if(trans.from != this->coord_frame || trans.to != dest)
182  {
183  printf("Cannot transform the source space using this coordinate transformation");//Consider throw
184  return false;
185  }
186 
187  MatrixXd t = trans.trans.block(0,0,3,4);
188 // res = src;
189  this->coord_frame = dest;
190  MatrixXd t_rr = MatrixXd::Ones(this->np, 4);
191  t_rr.block(0, 0, this->np, 3) = this->rr;
192  MatrixXd t_nn = MatrixXd::Zero(this->np, 4);
193  t_nn.block(0, 0, this->np, 3) = this->nn;
194 
195  this->rr = (t*t_rr.transpose()).transpose();
196  this->nn = (t*t_nn.transpose()).transpose();
197 
198  return true;
199 }
