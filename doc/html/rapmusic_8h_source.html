382 inline int RapMusic::getRank(const MatrixXT& p_matSigma)
383 {
384  int t_iRank;
385  //if once a singularvalue is smaller than epsilon = 10^-5 the following values are also smaller
386  // -> because Singular values are ordered
387  for(t_iRank = p_matSigma.rows()-1; t_iRank > 0; t_iRank--)
388  if (p_matSigma(t_iRank, t_iRank) > 0.00001)
389  break;
390 
391  t_iRank++;//rank corresponding to epsilon
392 
393  return t_iRank;
394 }
