206 void LayoutMaker::sphere_coord (float x,
207  float y,
208  float z,
209  float *r,
210  float *theta,
211  float *phi)
212 {
213  /* Rectangular to spherical coordinates */
214  float rxy = sqrt(x*x+y*y);
215  if (rxy < EPS) { /* Let's hope this is reasonable */
216  *r = z;
217  *theta = 0.0;
218  *phi = 0.0;
219  }
220  else {
221  *r = sqrt(x*x+y*y+z*z);
222  *theta = acos(z/(*r));
223  *phi = atan2 (y,x);
224  if (*phi < 0.0)
225  *phi = *phi + 2.0*M_PI;
226  }
227 }
