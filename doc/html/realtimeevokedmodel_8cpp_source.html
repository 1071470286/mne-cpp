588 void RealTimeEvokedModel::filterChannelsConcurrently()
589 {
590  //std::cout<<"START RealTimeEvokedModel::filterChannelsConcurrently()"<<std::endl;
591 
592  if(m_filterData.isEmpty())
593  return;
594 
595  //Generate QList structure which can be handled by the QConcurrent framework
596  QList<QPair<QList<FilterData>,QPair<int,RowVectorXd> > > timeData;
597  QList<int> notFilterChannelIndex;
598 
599  //Also append mirrored data in front and back to get rid of edge effects
600  for(qint32 i=0; i<m_matData.rows(); ++i) {
601  if(m_filterChannelList.contains(m_pRTE->info()->chs.at(i).ch_name)) {
602  RowVectorXd datTemp(m_matData.row(i).cols() + 2 * m_iMaxFilterLength);
603  datTemp << m_matData.row(i).head(m_iMaxFilterLength).reverse(), m_matData.row(i), m_matData.row(i).tail(m_iMaxFilterLength).reverse();
604  timeData.append(QPair<QList<FilterData>,QPair<int,RowVectorXd> >(m_filterData,QPair<int,RowVectorXd>(i,datTemp)));
605  }
606  else
607  notFilterChannelIndex.append(i);
608  }
609 
610  //Do the concurrent filtering
611  if(!timeData.isEmpty()) {
612  QFuture<void> future = QtConcurrent::map(timeData,
613  doFilterPerChannelRTE);
614 
615  future.waitForFinished();
616 
617  for(int r = 0; r<timeData.size(); r++)
618  m_matDataFiltered.row(timeData.at(r).second.first) = timeData.at(r).second.second.segment(m_iMaxFilterLength+m_iMaxFilterLength/2, m_matData.cols());
619  }
620 
621  //Fill filtered data with raw data if the channel was not filtered
622  for(int i = 0; i<notFilterChannelIndex.size(); i++)
623  m_matDataFiltered.row(notFilterChannelIndex.at(i)) = m_matData.row(notFilterChannelIndex.at(i));
624 
625  //std::cout<<"END RealTimeEvokedModel::filterChannelsConcurrently()"<<std::endl;
626 }
