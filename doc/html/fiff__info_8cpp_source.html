156 {
157  qDebug() << "make_compensator not debugged jet";

159  MatrixXd C1, C2, comp_tmp;
160 
161  qDebug() << "Todo add all need ctf variables.";

165 
166  if (from == to)

170  }
171 
172  if (from == 0)
173  C1 = MatrixXd::Zero(this->nchan,this->nchan);
174  else
175  {
176  if (!this->make_compensator(from, C1))
177  {
178  printf("Cannot create compensator C1\n");

180  return false;

182  }
183 
184  if (to == 0)
185  C2 = MatrixXd::Zero(this->nchan,this->nchan);
186  else
187  {
188  if (!this->make_compensator(to, C2))
189  {
190  printf("Cannot create compensator C2\n");
191  printf("Desired compensation matrix (kind = %d) not found\n",to);
192  return false;
193  }
194  }
195  //
196  // s_orig = s_from + C1*s_from = (I + C1)*s_from
197  // s_to = s_orig - C2*s_orig = (I - C2)*s_orig
198  // s_to = (I - C2)*(I + C1)*s_from = (I + C1 - C2 - C2*C1)*s_from
199  //
200  comp_tmp = MatrixXd::Identity(this->nchan,this->nchan) + C1 - C2 - C2*C1;
201 
202  qint32 k;
203  if (exclude_comp_chs)
204  {
205  VectorXi pick = MatrixXi::Zero(1,this->nchan);
206  qint32 npick = 0;
207  for (k = 0; k < this->nchan; ++k)

212  ++npick;
213  }
214  }
215  if (npick == 0)
216  {

218  return false;
219  }
220 
221  ctf_comp.data->data.resize(npick,this->nchan);
222  for (k = 0; k < npick; ++k)
223  ctf_comp.data->data.row(k) = comp_tmp.block(pick(k), 0, 1, this->nchan);
224  }
225  return true;

231 bool FiffInfo::make_compensator(fiff_int_t kind, MatrixXd& this_comp) const//private method
232 {
233  qDebug() << "make_compensator not debugged jet";
234  FiffNamedMatrix::SDPtr this_data;
235  MatrixXd presel, postsel;
236  qint32 k, col, c, ch, row, row_ch, channelAvailable;
237  for (k = 0; k < this->comps.size(); ++k)
238  {
239  if (this->comps[k].kind == kind)
240  { this_data = this->comps[k].data;
241  //
242  // Create the preselector
243  //
244  presel = MatrixXd::Zero(this_data->ncol,this->nchan);
245  for(col = 0; col < this_data->ncol; ++col)
246  {
247  channelAvailable = 0;
248  for (c = 0; c < this->ch_names.size(); ++c)
249  {
250  if (QString::compare(this_data->col_names.at(col),this->ch_names.at(c)) == 0)
251  {
252  ++channelAvailable;
253  ch = c;
254  }
255  }
256  if (channelAvailable == 0)
257  {
258  printf("Channel %s is not available in data\n",this_data->col_names.at(col).toUtf8().constData());
259  return false;
260  }
261  else if (channelAvailable > 1)
262  {
263  printf("Ambiguous channel %s",this_data->col_names.at(col).toUtf8().constData());
264  return false;
265  }
266  presel(col,ch) = 1.0;
267  }
268  //
269  // Create the postselector
270  //
271  postsel = MatrixXd::Zero(this->nchan,this_data->nrow);
272  for (c = 0; c < this->nchan; ++c)
273  {
274  channelAvailable = 0;
275  for (row = 0; row < this_data->row_names.size(); ++row)
276  {
277  if (QString::compare(this_data->col_names.at(c),this->ch_names.at(row)) == 0)
278  {
279  ++channelAvailable;
280  row_ch = row;
281  }
282  }
283  if (channelAvailable > 1)
284  {
285  printf("Ambiguous channel %s", this->ch_names.at(c).toUtf8().constData());
286  return false;
287  }
288  else if (channelAvailable == 1)
289  {
290  postsel(c,row_ch) = 1.0;
291  }
292  }
293  this_comp = postsel*this_data->data*presel;
294  return true;
295  }
296  }
297  this_comp = defaultMatrixXd;
298  return false;
299 }
