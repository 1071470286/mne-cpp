212  static bool make_compensator(FiffInfo* info, fiff_int_t from, fiff_int_t to, FiffCtfComp& ctf_comp, bool exclude_comp_chs = false)
213  {
214  qDebug() << "make_compensator not debugged jet";
215 
216  MatrixXf C1, C2, comp_tmp;
217 
218  qDebug() << "Todo add all need ctf variables.";
219  if(ctf_comp.data)
220  delete ctf_comp.data;
221  ctf_comp.data = new FiffNamedMatrix();
222 
223  if (from == to)
224  {
225  ctf_comp.data->data = MatrixXf::Zero(info->nchan, info->nchan);
226  return false;
227  }
228 
229  if (from == 0)
230  C1 = MatrixXf::Zero(info->nchan,info->nchan);
231  else
232  {
233  if (!make_compensator(info,from, C1))
234  {
235  printf("Cannot create compensator C1\n");
236  printf("Desired compensation matrix (kind = %d) not found\n",from);
237  return false;
238  }
239  }
240 
241  if (to == 0)
242  C2 = MatrixXf::Zero(info->nchan,info->nchan);
243  else
244  {
245  if (!make_compensator(info, to, C2))
246  {
247  printf("Cannot create compensator C2\n");
248  printf("Desired compensation matrix (kind = %d) not found\n",to);
249  return false;
250  }
251  }
252  //
253  // s_orig = s_from + C1*s_from = (I + C1)*s_from
254  // s_to = s_orig - C2*s_orig = (I - C2)*s_orig
255  // s_to = (I - C2)*(I + C1)*s_from = (I + C1 - C2 - C2*C1)*s_from
256  //
257  comp_tmp = MatrixXf::Identity(info->nchan,info->nchan) + C1 - C2 - C2*C1;
258 
259  qint32 k;
260  if (exclude_comp_chs)
261  {
262  VectorXi pick = MatrixXi::Zero(1,info->nchan);
263  qint32 npick = 0;
264  for (k = 0; k < info->nchan; ++k)
265  {
266  if (info->chs[k].kind != FIFFV_REF_MEG_CH)
267  {
268  pick(npick) = k;
269  ++npick;
270  }
271  }
272  if (npick == 0)
273  {
274  printf("Nothing remains after excluding the compensation channels\n");
275  return false;
276  }
277 
278  ctf_comp.data->data.resize(npick,info->nchan);
279  for (k = 0; k < npick; ++k)
280  ctf_comp.data->data.row(k) = comp_tmp.block(pick(k), 0, 1, info->nchan);
281  }
282  return true;

557

569  return false;

579  pos = events[0]->dir[k].pos;

582  FiffTag::read_tag(t_pFile,t_pTag,pos);
583  if(t_pTag->type == FIFFT_UINT)

594  delete t_pFile;
595  delete t_pTree;
596  delete t_pDir;
597  delete t_pTag;
598  printf("Could not find any events\n");
