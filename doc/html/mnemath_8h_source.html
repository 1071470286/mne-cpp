313 template< typename T>
314 VectorXi MNEMath::sort(Matrix<T, Dynamic, 1> &v, bool desc)
315 {
316  std::vector< std::pair<int,T> > t_vecIdxValue;
317  VectorXi idx(v.size());
318 
319  if(v.size() > 0)
320  {
321  //fill temporal vector
322  for(qint32 i = 0; i < v.size(); ++i)
323  t_vecIdxValue.push_back(std::pair<int,T>(i, v[i]));
324 
325  //sort temporal vector
326  if(desc)
327  std::sort(t_vecIdxValue.begin(), t_vecIdxValue.end(), MNEMath::compareIdxValuePairBiggerThan<T>);
328  else
329  std::sort(t_vecIdxValue.begin(), t_vecIdxValue.end(), MNEMath::compareIdxValuePairSmallerThan<T>);
330 
331  //store results
332  for(qint32 i = 0; i < v.size(); ++i)
333  {
334  idx[i] = t_vecIdxValue[i].first;
335  v[i] = t_vecIdxValue[i].second;
336  }
337  }
338 
339  return idx;
340 }
