118 MNEInverseOperator MNEInverseOperator::prepare_inverse_operator(qint32 nave ,float lambda2, bool dSPM, bool sLORETA)
119 {
120  if(nave <= 0)
121  {
122  printf("The number of averages should be positive\n");
123  return MNEInverseOperator();
124  }
125  printf("Preparing the inverse operator for use...\n");
126  MNEInverseOperator inv(*this);
127  //
128  // Scale some of the stuff
129  //
130  float scale = ((float)inv.nave)/((float)nave);
131  inv.noise_cov->data *= scale;
132  inv.noise_cov->eig *= scale;
133  inv.source_cov->data *= scale;
134  //
135  if (inv.eigen_leads_weighted)
136  inv.eigen_leads->data *= sqrt(scale);
137  //
138  printf("\tScaled noise and source covariance from nave = %d to nave = %d\n",inv.nave,nave);
139  inv.nave = nave;
140  //
141  // Create the diagonal matrix for computing the regularized inverse
142  //
143  VectorXd tmp = inv.sing.cwiseProduct(inv.sing) + VectorXd::Constant(inv.sing.size(), lambda2);
144 // if(inv.reginv)
145 // delete inv.reginv;
146  inv.reginv = VectorXd(inv.sing.cwiseQuotient(tmp));
147  printf("\tCreated the regularized inverter\n");
148  //
149  // Create the projection operator
150  //
151 
152  qint32 ncomp = FiffInfo::make_projector(inv.projs, inv.noise_cov->names, inv.proj);
153  if (ncomp > 0)
154  printf("\tCreated an SSP operator (subspace dimension = %d)\n",ncomp);
155 
156  //
157  // Create the whitener
158  //
159 // if(inv.whitener)
160 // delete inv.whitener;
161  inv.whitener = MatrixXd::Zero(inv.noise_cov->dim, inv.noise_cov->dim);
162 
163  qint32 nnzero, k;
164  if (inv.noise_cov->diag == 0)
165  {
166  //
167  // Omit the zeroes due to projection
168  //
169  nnzero = 0;
170 
171  for (k = ncomp; k < inv.noise_cov->dim; ++k)
172  {
173  if (inv.noise_cov->eig[k] > 0)
174  {
175  inv.whitener(k,k) = 1.0/sqrt(inv.noise_cov->eig[k]);
176  ++nnzero;
177  }
178  }
179  //
180  // Rows of eigvec are the eigenvectors
181  //
182  inv.whitener *= inv.noise_cov->eigvec;
183  printf("\tCreated the whitener using a full noise covariance matrix (%d small eigenvalues omitted)\n", inv.noise_cov->dim - nnzero);
184  }
185  else
186  {
187  //
188  // No need to omit the zeroes due to projection
189  //
190  for (k = 0; k < inv.noise_cov->dim; ++k)
191  inv.whitener(k,k) = 1.0/sqrt(inv.noise_cov->data(k,0));
192 
193  printf("\tCreated the whitener using a diagonal noise covariance matrix (%d small eigenvalues discarded)\n",ncomp);
194  }
195  //
196  // Finally, compute the noise-normalization factors
197  //
198  if (dSPM || sLORETA)
199  {
200  VectorXd* noise_norm = new VectorXd(VectorXd::Zero(inv.eigen_leads->nrow));
201  VectorXd* noise_weight = NULL;
202  if (dSPM)
203  {
204  printf("\tComputing noise-normalization factors (dSPM)...");
205  noise_weight = new VectorXd(inv.reginv);
206  }
207  else
208  {
209  printf("\tComputing noise-normalization factors (sLORETA)...");
210  VectorXd tmp = (VectorXd::Constant(inv.sing.size(), 1) + inv.sing.cwiseProduct(inv.sing)/lambda2);
211  noise_weight = new VectorXd(inv.reginv.cwiseProduct(tmp.cwiseSqrt()));
212  }
213  VectorXd one;
214  if (inv.eigen_leads_weighted)
215  {
216  for (k = 0; k < inv.eigen_leads->nrow; ++k)
217  {
218  one = inv.eigen_leads->data.block(k,0,1,inv.eigen_leads->data.cols()).cwiseProduct(*noise_weight);
219  (*noise_norm)[k] = sqrt(one.dot(one));
220  }
221  }
222  else
223  {
224  float c;
225  for (k = 0; k < inv.eigen_leads->nrow; ++k)
226  {
227  c = sqrt(inv.source_cov->data(k,0));
228  one = c*(inv.eigen_leads->data.block(k,0,1,inv.eigen_leads->data.cols()).transpose()).cwiseProduct(*noise_weight);//ToDo eigenleads data -> pointer
229  (*noise_norm)[k] = sqrt(one.dot(one));
230  }
231  }
232  //
233  // Compute the final result
234  //
235  VectorXd noise_norm_new;
236  if (inv.source_ori == FIFFV_MNE_FREE_ORI)
237  {
238  //
239  // The three-component case is a little bit more involved
240  // The variances at three consequtive entries must be squeared and
241  // added together
242  //
243  // Even in this case return only one noise-normalization factor
244  // per source location
245  //
246  VectorXd* t = MNEMath::combine_xyz(noise_norm->transpose());
247  noise_norm_new = t->cwiseSqrt();//double otherwise values are getting too small
248  delete t;
249  //
250  // This would replicate the same value on three consequtive
251  // entries
252  //
253  // noise_norm = kron(sqrt(mne_combine_xyz(noise_norm)),ones(3,1));
254  }
255  VectorXd vOnes = VectorXd::Ones(noise_norm_new.size());
256  VectorXd tmp = vOnes.cwiseQuotient(noise_norm_new.cwiseAbs());
257 // if(inv.noisenorm)
258 // delete inv.noisenorm;
259  inv.noisenorm = SparseMatrix<double>(noise_norm_new.size(),noise_norm_new.size());
260 
261  for(qint32 i = 0; i < noise_norm_new.size(); ++i)
262  inv.noisenorm.insert(i,i) = tmp[i];
263 
264  delete noise_weight;
265  delete noise_norm;
266  printf("[done]\n");
267  }
268  else
269  {
270 // if(inv.noisenorm)
271 // delete inv.noisenorm;
272  inv.noisenorm = SparseMatrix<double>();
273  }
274 
275  return inv;
276 }
